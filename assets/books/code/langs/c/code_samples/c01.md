# printf

```c
#include <stdio.h>
void main(void) {
    int i = 1;
    unsigned u = 2;
    long l = 3;
    float f = 4.0;
    double d = 5.0;
    char c = 6;
    unsigned char uc = 7;
    printf("i = %d, u = %u, l = %l, f = %f, d = %lf, c = %c, c = %d, uc = %d\n",
            i, u, l, f, d, c, c, uc);
    printf("print a tab by \t and new line by \n");
}
```

# Math Functions

```c
#include <math.h>

void main(void) {
    double th = pi/2;    // th is in radians
    double x, y;

    x = cos(th);
    y = sin(th);
    th = atan2(y, x);
}
```
